% Microquestions
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newcounter{microquestioncounter}
\counterwithin*{microquestioncounter}{subsection}
\newcommand{\q}[1]{
  \stepcounter{microquestioncounter} 
  \begin{quote} 
    \textbf{
      Q\thesubsection-
      \Roman{microquestioncounter}:
    } 
    {#1}? 
  \end{quote} 
}

\newcommand{\qs}[2]{
  \stepcounter{microquestioncounter} 
  \begin{quote} 
    \textbf{
      Q\thesubsection-
      \Roman{microquestioncounter}:
    } 
    {#1}? 
    \par  
    \textbf{sources:} 
    {#2} 
  \end{quote} 
}
